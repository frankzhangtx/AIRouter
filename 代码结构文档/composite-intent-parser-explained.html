class CompositeIntentParser(
    private val ruleBasedIntentParser: IntentParser,
    private val llmIntentParser: IntentParser,
    private val config: IntentParsingConfig
) : IntentParser {
    override suspend fun parse(request: ParseRequest): ParseOutcome {
        // 第一步永远先走规则解析器，拿到一个 ruleOutcome。
        val ruleOutcome = ruleBasedIntentParser.parse(request)

        // 只有同时满足 3 个条件，才直接信任规则结果并立刻返回：
        // 1. 规则解析成功
        // 2. 目标不是 Unknown，说明不是“我没识别出来”
        // 3. 置信度达到配置阈值 minRuleConfidence
        if (ruleOutcome is ParseOutcome.Success &&
            ruleOutcome.result.userGoal != UserGoal.Unknown &&
            ruleOutcome.result.confidence >= config.minRuleConfidence
        ) {
            return ruleOutcome
        }

        // 如果规则结果不够强，但配置明确禁止 LLM，再差也只能把规则结果原样返回。
        if (!config.enableLlmParsing) {
            return ruleOutcome
        }

        // 允许 LLM 时，再发起第二阶段解析。
        val llmOutcome = llmIntentParser.parse(request)

        // 只要 LLM 成功，就优先采用 LLM 的结构化结果。
        if (llmOutcome is ParseOutcome.Success) {
            // 这里会重写 parserSource，用来保留“这次结果是如何得到的”。
            val parserSource = if (ruleOutcome is ParseOutcome.Success) {
                // 说明规则其实也产出了结果，只是置信度不够，后来被 LLM 接管。
                ParserSource.RULE_THEN_LLM
            } else {
                // 说明规则阶段直接失败，最终结果纯粹来自 LLM。
                ParserSource.LLM
            }
            return ParseOutcome.Success(
                llmOutcome.result.copy(
                    parserMetadata = llmOutcome.result.parserMetadata.copy(
                        parserSource = parserSource
                    )
                )
            )
        }

        // 走到这里说明 LLM 失败了。
        return if (config.fallbackToRuleOnLlmFailure && ruleOutcome is ParseOutcome.Success) {
            // 如果允许失败兜底，而且规则阶段其实拿到过一个成功结果，
            // 那就把那个结果救回来，并把来源标记成 FALLBACK。
            ParseOutcome.Success(
                ruleOutcome.result.copy(
                    parserMetadata = ruleOutcome.result.parserMetadata.copy(
                        parserSource = ParserSource.FALLBACK
                    )
                )
            )
        } else {
            // 否则就把 LLM 的失败直接往上抛，交给上层处理。
            llmOutcome
        }
    }
}
这个类不是把两个解析器的结果做加权投票，而是明显的串行策略： Rule First - LLM Second - Optional Rule Fallback。 真正决定是否短路、是否继续、是否回退的控制点，全部集中在 minRuleConfidence、enableLlmParsing 和 fallbackToRuleOnLlmFailure 这三个配置上。

阅读抓手

先抓住这 4 个判断，再回头看代码，逻辑会很直。

1. 高置信规则结果可短路

规则结果必须成功、不是 UserGoal.Unknown、并且置信度达到阈值，才会直接返回；否则就视作“还不能信”。

2. 是否允许 LLM 是总开关

enableLlmParsing 关闭时，类不会尝试第二阶段解析，即使规则结果只是低置信成功或完全失败。

3. LLM 成功后会改写来源标签

当 LLM 接管结果时，parserMetadata.parserSource 不一定保留原值，而是被改成 RULE_THEN_LLM 或 LLM。

4. 回退只在一种场景成立

必须同时满足“LLM 失败”“开启回退”“规则阶段曾成功”三项条件，才会返回 FALLBACK 结果。

RULE：规则阶段直接命中并短路
RULE_THEN_LLM：规则有结果但不够稳，最终被 LLM 接管
LLM：规则阶段失败，结果纯来自 LLM
FALLBACK：LLM 失败后退回规则结果

关键分支与测试对照

测试文件把这个类的主路径基本全覆盖了，适合和源码一起看。

规则命中并短路返回

对应测试 parse_returnsRuleResultWhenConfidenceExceedsThreshold()。 当规则结果置信度达到 0.82f 阈值时，LLM 即使也能返回结果，也根本不会被采用。

规则低置信，交给 LLM 接管

对应测试 parse_usesLlmWhenRuleConfidenceFallsBelowThreshold()。 规则成功但只有 0.5f，所以继续调用 LLM，最终来源标签改成 RULE_THEN_LLM。

LLM 失败，回退到规则结果

对应测试 parse_returnsRuleFallbackWhenLlmFailsAndFallbackEnabled()。 这里规则结果虽然低置信，但仍是一个成功结果，因此可以在 LLM 超时后被救回来。

LLM 失败，失败直接上抛

对应测试 parse_returnsLlmFailureWhenFallbackDisabled()。 当 fallbackToRuleOnLlmFailure 为 false 时，最终返回的就是 LLM 的 ParseError.Timeout。

一个容易忽略的点是：低置信规则结果并不会被立刻丢弃。它会暂存在 ruleOutcome 里，既用于判定 LLM 成功时应标记为 RULE_THEN_LLM，也用于 LLM 失败时是否能够执行 FALLBACK。

执行流程图

从进入 parse(request) 开始，到返回最终 ParseOutcome 为止。

整个流程的关键不是“谁先返回”，而是“规则结果是否足够可信”。一旦不足够可信， CompositeIntentParser 就把规则结果保留下来，既作为 LLM 接管时的上下文， 也作为 LLM 失败时的最后兜底。
